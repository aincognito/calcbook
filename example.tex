\begin{question}
  Which one will you choose?
  \begin{multipleChoice}
    \choice[correct]{I'm correct.}
    \choice{I'm wrong.}
    \choice{I'm wrong too.}
  \end{multipleChoice}
\end{question}

\begin{question}
  Which one will you choose?
  \begin{selectAll}
    \choice[correct]{I'm correct.}
    \choice{I'm wrong.}
    \choice[correct]{I'm also correct.}
    \choice{I'm wrong too.}
  \end{selectAll}
\end{question}
